const burgerButton = document.querySelector('.header__burger');
const nav = document.querySelector('.header__nav');

if (burgerButton && nav) {
  burgerButton.addEventListener('click', () => {
    const isOpen = nav.classList.toggle('nav--open');
    burgerButton.classList.toggle('header__burger--open', isOpen);
    burgerButton.setAttribute('aria-expanded', String(isOpen));
  });

  nav.querySelectorAll('.nav__link').forEach((link) => {
    link.addEventListener('click', () => {
      nav.classList.remove('nav--open');
      burgerButton.classList.remove('header__burger--open');
      burgerButton.setAttribute('aria-expanded', 'false');
    });
  });

  window.addEventListener('resize', () => {
    if (window.innerWidth >= 480) {
      nav.classList.remove('nav--open');
      burgerButton.classList.remove('header__burger--open');
      burgerButton.setAttribute('aria-expanded', 'false');
    }
  });
}

У чому проблема: В атрибут src не можна дописувати дескриптор щільності екрана (1x). Браузер сприйме фразу "./imgs/comand.png 1x" як єдине ім'я файлу, не знайде картинку з пробілом у назві й видасть помилку завантаження 404.

Як виправити: В src залишайте лише чистий шлях: src="./imgs/comand.png".

Зламаний тег <picture> у секції відео
HTML
<picture class="video-section__bg">
  <source srcset="./imgs/comand.png 1x, ./imgs/comand@2x.png 2x" />
  <img src="./imgs/comand.png" ... />
</picture>
У чому проблема: Тег <source> всередині <picture> обов'язково повинен мати атрибут media (наприклад, media="(max-width: 768px)") або атрибут type (наприклад, type="image/webp"). Якщо ви просто хочете дати ретина-картинки (1x/2x) без зміни формату чи адаптиву, тег <picture> взагалі не потрібен. Достатньо одного тегу <img> з атрибутом srcset.

Додатково: Рядок <source srcset="./imgs/city.png 1x, ./imgs/city@2.png 2x" /> містить друкарську помилку в розширенні (city@2.png замість @2x).

2. Порушення семантики та доступності (Accessibility)
Кнопка без тексту (Бургер-меню)
HTML
<button class="header__burger" ...>
  <span></span>
  <span></span>
  <span></span>
</button>
Ви чудово використали атрибути aria-label та aria-controls. Це плюс. Але якщо JS не завантажиться, або для деяких скрінрідерів, три порожні теги <span> всередині кнопки є абсолютно нечитабельними.

Порада: Краще додати всередину прихований текст для роботів: <span class="visually-hidden">Меню</span>.

Семантичний хаос у футері
Колонки футера загорнуті в теги заголовків <h4>. При цьому на сторінці немає жодного <h5> чи чіткої ієрархічної структури вище, яка б виправдовувала саме четвертий рівень. Самі посилання згруповані в списки <ul>, що правильно, але адреса в першій колонці:

HTML
<li class="main-footer__item main-footer__item--contact">
  <span class="main-footer__text">Тернопіль вул С. Бандери</span>
</li>
Як виправити: Контактну інформацію (телефон, пошту, адресу) у футері завжди варто загортати в семантичний тег <address>.


Резюме
Каркас сторінки непоганий, інтеграція бібліотеки анімацій AOS прописана акуратно. Проте проєкт потребує виправлення помилок із src/srcset, які буквально ламають відображення картинок у браузері, та фінальної чистки контенту.

1. CSS: Архітектурні помилки та баги
Критичний баг: "Зламаний" брейкпоінт 480px та накладання стилів
В медіа-запитах є логічна помилка, через яку приховане меню на мобільних пристроях працюватиме некоректно.

У чому проблема: Ви ховаєте бургер-меню у медіа-запиті @media (max-width: 479px). Але наступний запит написаний як @media (min-width: 480px). Якщо ширина екрана буде рівно 479.5px (що часто буває на Retina-дисплеях через піксельне масштабування), жоден із цих запитів може не спрацювати належним чином, або вони накладуться.

Баг у JS: Ваша перевірка window.innerWidth >= 480 працює з фізичними пікселями вікна без урахування смуги прокрутки.

Як виправити: Замість створення "ізольованих" проміжків на кшталт max-width: 479px, залиште базові стилі для мобільної версії у загальному просторі (Mobile First), а для десктопу використовуйте виключно min-width без дублювання.

Порушення порядку медіа-запитів (Каскадність)
У кінці файлу ваші медіа-запити йдуть у такому порядку:

@media (min-width: 768px)

@media (max-width: 479px)

@media (min-width: 480px)

@media (min-width: 1280px)

@media (min-width: 1024px)

У чому проблема: Це порушує логіку висхідного каскаду. Запит min-width: 1024px стоїть після min-width: 1280px. Через це правила з 1024px переб'ють (перезапишуть) специфічні стилі для великих екранів 1280px, які написані вище.

Правильний порядок (Mobile First):

CSS
/* Базові мобільні стилі */
@media (min-width: 480px) { ... }
@media (min-width: 768px) { ... }
@media (min-width: 1024px) { ... }
@media (min-width: 1280px) { ... }
Дублювання селекторів (Копіпаст)
У коді двічі підряд оголошено один і той самий селектор із різними властивостями:

CSS
.subscribe-section {
  background-color: var(--color-surface);
  position: relative;
  z-index: 1;
  padding-top: 250px;
  padding-bottom: 80px;
  text-align: center;
}
.subscribe-section {
  font-family: "Helvetica Neue", Arial, sans-serif;
  background-color: var(--color-surface);
  color: var(--color-text-dark);
  padding: 60px 0;
  text-align: center;
}
У чому проблема: Друге оголошення повністю затирає padding-top: 250px (змінюючи його на 60px). Це ламає задуманий дизайн-макет, де, очевидно, відступ мав компенсувати зміщення відеоплеєра (transform: translateY(115px)).

Жорстко закодовані кольори (Hardcoded values)
Ви створили чудову палітру змінних у :root, але в секції .portfolio__purple-bg використали прямий градієнт: linear-gradient(90deg, #4d57c8 0%, #6316df 100%).

Порада: Винесіть ці кольори в змінні (наприклад, --color-brand-start, --color-brand-end), щоб у разі ребрендингу сайту колір мінявся в одному місці.

2. JavaScript: Якість коду та оптимізація
Код написаний грамотно, але подія resize містить класичну помилку продуктивності.

Проблема з Performance (resize event)
JavaScript
window.addEventListener('resize', () => {
  if (window.innerWidth >= 480) { ... }
});
У чому проблема: Подія resize спрацьовує десятки разів за секунду, коли користувач просто змінює розмір вікна чи повертає екран смартфона. Виконання важких операцій з DOM (classList.remove) на кожен мікро-піксель навантажує процесор (особливо на мобільних пристроях).

Як виправити: Замість resize для відстеження брейкпоінтів у JS набагато краще використовувати сучасний інструмент window.matchMedia. Він працює як CSS медіа-запит і викликає колбек тільки один раз — у момент переходу через межу екрана.

JavaScript
const mediaQuery = window.matchMedia('(min-width: 480px)');

function handleTabletChange(e) {
  if (e.matches) {
    nav.classList.remove('nav--open');
    burgerButton.classList.remove('header__burger--open');
    burgerButton.setAttribute('aria-expanded', 'false');
  }
}

// Слухаємо зміну стану екрана
mediaQuery.addEventListener('change', handleTabletChange);

Сильні сторони:
Чудова структура іменування класів за БЕМ.

Професійна робота зі змінними для організації тем оформлення та кольорів.

Логічний JS без використання "важких" бібліотек на кшталт jQuery.

Прибрати дескриптор 1x з атрибута src в HTML (помилка 404 для картинок).

Поставити CSS медіа-запити в строгому порядку зростання (min-width), прибравши хаотичні max-width.

Оптимізувати JS за допомогою matchMedia замість "прожорливого" resize.

1. Семантичні помилки та валідація
 <section class="logo_nav"> всередині <header>
Проблема: Тег <section> обов'язково повинен мати заголовок (<h2>-<h6>). Крім того, групувати логотип та навігацію в окрему секцію всередині шапки — семантично неправильно.

Як виправити: Замініть <section class="logo_nav"> на звичайний <div class="header__top"> або <div class="header__wrapper">.

Кілька головних заголовків <h1>
У коді є два теги <h1>: один у Hero-секції (EXPLORE INDONESIA), інший у галереї (Photo Gallery).

Проблема: За стандартами SEO та доступності (WCAG), на сторінці має бути лише один заголовок <h1>, який описує суть усього документа.

Як виправити: Заголовок у галереї <h1>Photo Gallery</h1> потрібно замінити на <h2>Photo Gallery</h2>.

Несемантичні кнопки та іконки
У блоці соцмереж футера та шапки іконки та посилання зроблені просто текстом у тегах <p> або порожніми посиланнями:

Проблема: Скрінрідери для людей із вадами зору прочитають це просто як одинокі літери "еф", "ін", "ге". Користувач не зрозуміє, що це кнопки соцмереж.

Як виправити: Використовувати список <ul>, всередині посилання <a>, а текст приховувати за допомогою класу visually-hidden (або додавати aria-label).

HTML
<li>
  <a href="#" aria-label="Facebook">
    <svg>...</svg>

Ви використали цікавий прийом, щоб приховати зайвий контент на мобільних екранах:

⚠️ Зверніть увагу: Якщо у браузері користувача буде увімкнено збільшений шрифт або дефолтна висота рядка (line-height) перевищить 22px, текст обріжеться некрасиво прямо посередині літер. ⚡ Рев’ю JavaScript
1. mobile_menu.js
Неймовірно чистий, модульний та продуктивний код завдяки роботі з classList. Чудове рішення з використанням стандартних нативних DOM API.

2. modal.js (Рушій каруселі)
Ваша логіка обчислення слайдера чудово керує конфігураціями з кількома слайдами під час зміни розміру вікна. Проте є один нюанс із таймінгом обчислення slideWidth():

JavaScript
function slideWidth() {
  const slide = track.querySelector(".carousel__slide");
  return slide.getBoundingClientRect().width + gap();
}
У чому підвох: Якщо функція slideWidth() викликається миттєво під час завантаження сторінки, CSS-стилі або зображення можуть ще не встигнути повністю завантажитися. Через це getBoundingClientRect().width поверне 0 або неточне субпіксельне значення.

Рішення: Перенесіть ініціалізацію та виконання коду всередину слухача подій load або DOMContentLoaded, або ж безпечно обчислюйте метрики розмітки безпосередньо всередині виконання кроку goTo().

Оптимізований та стабільний скрипт каруселі:
JavaScript
document.addEventListener("DOMContentLoaded", () => {
  const track = document.getElementById("carouselTrack");
  const dotsEl = document.getElementById("carouselDots");
  const btnPrev = document.getElementById("prev");
  const btnNext = document.getElementById("next");

  if (!track || !dotsEl) return; // Захист від помилок, якщо елементів немає в DOM

  const total = track.querySelectorAll(".carousel__slide").length;
  let current = 0;

  const visibleSlides = () => window.innerWidth <= 480 ? 1 : 3;
  const gap = () => window.innerWidth <= 480 ? 0 : 16;
  const getSteps = () => total - visibleSlides() + 1;

  function slideWidth() {
    const slide = track.querySelector(".carousel__slide");
    return slide ? slide.getBoundingClientRect().width + gap() : 0;
  }

  function createDots() {
    dotsEl.innerHTML = "";
    const steps = getSteps();
    for (let i = 0; i < steps; i++) {
      const dot = document.createElement("button");
      dot.className = "carousel__dot";
      if (i === current) dot.classList.add("active");
      dot.addEventListener("click", () => goTo(i));
      dotsEl.appendChild(dot);
    }
  }

  function goTo(index) {
    const steps = getSteps();
    if (index < 0) index = steps - 1;
    if (index >= steps) index = 0;

    current = index;
    track.style.transform = `translateX(-${current * slideWidth()}px)`;

    document.querySelectorAll(".carousel__dot").forEach((dot, i) => {
      dot.classList.toggle("active", i === current);
    });
  }

  btnPrev.addEventListener("click", () => goTo(current - 1));
  btnNext.addEventListener("click", () => goTo(current + 1));

  // Оптимізація зміни розміру екрану через Debounce (таймаут)
  let resizeTimeout;
  window.addEventListener("resize", () => {
    clearTimeout(resizeTimeout);
    resizeTimeout = setTimeout(() => {
      const steps = getSteps();
      if (current > steps - 1) current = steps - 1;
      createDots();
      goTo(current);
    }, 100);
  });

  // Безпечний запуск ініціалізації
  createDots();
  goTo(0);
});
